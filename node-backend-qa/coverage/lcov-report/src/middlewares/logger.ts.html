import { MiddlewareHandler } from "hono";
 
export const logger: MiddlewareHandler = async (c, next) => {
  const start = Date.now();
  const method = c.req.method;
  const url = c.req.url;
  const path = new URL(url).pathname;
  
  // Log request start
  console.log(`\nüì• ${method} ${path} - Request started`);
  
  try {
    // Process the request
    await next();
    
    // Calculate response time
    const end = Date.now();
    const responseTime = end - start;
    const status = c.res.status;
    
    // Determine log level based on status code
    let logLevel = "‚úÖ";
    if (status >= 400 && status < 500) {
      logLevel = "‚ö†Ô∏è";
    } else if (status >= 500) {
      logLevel = "‚ùå";
    }
    
    // Log response
    console.log(`${logLevel} ${method} ${path} - ${status} (${responseTime}ms)`);
    
    // Log additional info for errors
    if (status >= 400) {
      console.log(`   Error: ${c.res.statusText || "Unknown error"}`);
    }
    
  } catch (error) {
    const end = Date.now();
    const responseTime = end - start;
    
    console.error(`‚ùå ${method} ${path} - Error (${responseTime}ms)`);
    console.error(`   Error: ${error instanceof Error ? error.message : String(error)}`);
    
    // Re-throw the error to maintain the error handling chain
    throw error;
  }
};
